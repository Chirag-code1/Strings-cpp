/Convert to lower case:
#include<iostream>
#include<string>
#include<algorithm>

using namespace std;
int main(){
    string str="HGCHSjan";
    
   
 
   
   //covert to lower case:
   for(int i=0;i<str.size();i++){
       if(str[i]>='A'&&str[i]<='Z'){
           str[i]+=32;
       }
   }
   
   cout<<"By itetrative method: "+str<<endl;
   
   
   transform(str.begin(),str.end(),str.begin(), ::tolower);
   cout<<"By inbuilt function: "+str<< endl;
   
    
    return 0;
}
